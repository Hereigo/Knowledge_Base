# EXAMPLES :
```sql

---- DATE-TIME FORMAT :

PRINT CONVERT(datetime, CONVERT (varchar(10), GETDATE(), 105), 105);
-- >> Feb 17 2022 12:00AM

---- FIND FUNCTION OR PROCEDURE :

DECLARE @SEARCHING_PATTERN varchar(50) = '%' + 'User' + '%'; -- (for example)
SELECT TOP 10 * FROM INFORMATION_SCHEMA.TABLES WHERE TABLE_NAME LIKE @SEARCHING_PATTERN;
SELECT TOP 10 LAST_ALTERED, ROUTINE_SCHEMA, ROUTINE_TYPE, ROUTINE_NAME, *
    FROM INFORMATION_SCHEMA.ROUTINES WHERE ROUTINE_NAME LIKE @SEARCHING_PATTERN;

---- TO SEE THE BODY OF SELECTED FUNCTION OR PROCEDURE :

DECLARE @SEARCHING_PATTERN varchar(50) = '%' + 'User' + '%'; -- (for example)
DECLARE @x varchar(max);
SET @x = (SELECT TOP 1 ROUTINE_DEFINITION
	FROM INFORMATION_SCHEMA.ROUTINES
	WHERE ROUTINE_NAME LIKE @SEARCHING_PATTERN);
PRINT @x;

---- FIND BY NAME WHERE COLUMN EXISTS :
DECLARE @SEARCHING_PATTERN varchar(50) = '%' + 'User' + '%'; -- (for example)
SELECT COLUMN_NAME, TABLE_NAME
FROM INFORMATION_SCHEMA.COLUMNS
WHERE COLUMN_NAME LIKE @SEARCHING_PATTERN;

```
